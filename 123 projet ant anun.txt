Concept

A student logs:

😊 Mood (1–10 or emoji scale)
📚 Study time
📱 Screen time
🏃 Activities (exercise, gaming, socializing, reading, etc.)
✅ Daily productivity score (or completed tasks)

The website then analyzes patterns and answers questions like:

"Do I study better when I'm happy?"
"Does more screen time reduce my productivity?"
"Does exercise improve my focus?"
"Which activities help me have productive days?"

server.js
const express = require("express");
const cors = require("cors");
const sqlite3 = require("sqlite3").verbose();

const app = express();
app.use(cors());
app.use(express.json());

const db = new sqlite3.Database("./tracker.db");

db.serialize(() => {
  db.run(`
    CREATE TABLE IF NOT EXISTS entries (
      id INTEGER PRIMARY KEY AUTOINCREMENT,
      mood INTEGER,
      studyTime REAL,
      screenTime REAL,
      productivity INTEGER,
      activity TEXT,
      date TEXT
    )
  `);
});

// Add entry
app.post("/entries", (req, res) => {
  const {
    mood,
    studyTime,
    screenTime,
    productivity,
    activity,
    date,
  } = req.body;

  db.run(
    `INSERT INTO entries
     (mood, studyTime, screenTime, productivity, activity, date)
     VALUES (?, ?, ?, ?, ?, ?)`,
    [mood, studyTime, screenTime, productivity, activity, date],
    function (err) {
      if (err) return res.status(500).json(err);
      res.json({ id: this.lastID });
    }
  );
});

// Get all entries
app.get("/entries", (req, res) => {
  db.all("SELECT * FROM entries", [], (err, rows) => {
    if (err) return res.status(500).json(err);
    res.json(rows);
  });
});

app.listen(5000, () => {
  console.log("Server running on port 5000");
});

Run:

node server.js

App.jsx
import { useState, useEffect } from "react";
import axios from "axios";
import {
  LineChart,
  Line,
  XAxis,
  YAxis,
  CartesianGrid,
  Tooltip
} from "recharts";

function App() {
  const [entries, setEntries] = useState([]);

  const [form, setForm] = useState({
    mood: "",
    studyTime: "",
    screenTime: "",
    productivity: "",
    activity: "",
  });

  useEffect(() => {
    fetchEntries();
  }, []);

  const fetchEntries = async () => {
    const res = await axios.get("http://localhost:5000/entries");
    setEntries(res.data);
  };

  const submit = async () => {
    await axios.post("http://localhost:5000/entries", {
      ...form,
      date: new Date().toLocaleDateString(),
    });

    fetchEntries();

    setForm({
      mood: "",
      studyTime: "",
      screenTime: "",
      productivity: "",
      activity: "",
    });
  };

  return (
    <div style={{ padding: 30 }}>
      <h1>Student Productivity Tracker</h1>

      <input
        placeholder="Mood (1-10)"
        value={form.mood}
        onChange={(e) =>
          setForm({ ...form, mood: e.target.value })
        }
      />

      <input
        placeholder="Study Hours"
        value={form.studyTime}
        onChange={(e) =>
          setForm({ ...form, studyTime: e.target.value })
        }
      />

      <input
        placeholder="Screen Time"
        value={form.screenTime}
        onChange={(e) =>
          setForm({ ...form, screenTime: e.target.value })
        }
      />

      <input
        placeholder="Productivity %"
        value={form.productivity}
        onChange={(e) =>
          setForm({
            ...form,
            productivity: e.target.value,
          })
        }
      />

      <input
        placeholder="Activity"
        value={form.activity}
        onChange={(e) =>
          setForm({
            ...form,
            activity: e.target.value,
          })
        }
      />

      <button onClick={submit}>
        Save Entry
      </button>

      <h2>Productivity Trend</h2>

      <LineChart
        width={700}
        height={300}
        data={entries}
      >
        <CartesianGrid strokeDasharray="3 3" />
        <XAxis dataKey="date" />
        <YAxis />
        <Tooltip />
        <Line
          type="monotone"
          dataKey="productivity"
          stroke="#8884d8"
        />
      </LineChart>
    </div>
  );
}

export default App;
AI Insights Function

Add this utility:

function generateInsight(entries) {
  if (entries.length < 5)
    return "Need more data.";

  const avgMood =
    entries.reduce(
      (sum, e) => sum + Number(e.mood),
      0
    ) / entries.length;

  const avgProductivity =
    entries.reduce(
      (sum, e) => sum + Number(e.productivity),
      0
    ) / entries.length;

  return `
  Your average mood is ${avgMood.toFixed(1)}.
  Average productivity is ${avgProductivity.toFixed(1)}%.
  `;
}
